```rust
use std::path::{Path, PathBuf};
use lopdf::{Document, Dictionary, Object};
use tauri::AppHandle;

use crate::error::{AppError, Result};
use crate::pipeline::{
    temp::TempStage,
    progress::{emit_progress, emit_complete, emit_error},
};
use crate::tools::ProcessRequest;

#[derive(Debug, serde::Deserialize, Default)]
pub struct MetadataOptions {
    pub title: Option<String>,
    pub author: Option<String>,
    pub subject: Option<String>,
    pub keywords: Option<String>,
    pub creator: Option<String>,
}

/// Apply metadata fields to a document. Returns new Document (immutable pattern).
/// `None` = leave existing value. `Some("")` = remove field.
pub fn apply_metadata(mut doc: Document, opts: &MetadataOptions) -> Result<Document> {
    // Get or create Info dictionary
    let info_id = match doc.trailer.get(b"Info").and_then(|o| o.as_reference().ok()) {
        Some(id) => id,
        None => {
            let id = doc.add_object(Object::Dictionary(Dictionary::new()));
            doc.trailer.set("Info", Object::Reference(id));
            id
        }
    };

    let info = doc
        .get_object_mut(info_id)
        .map_err(|e| AppError::Pdf(format!("Cannot access Info dict: {e}")))?
        .as_dict_mut()
        .map_err(|e| AppError::Pdf(format!("Info is not a dictionary: {e}")))?;

    fn apply_field(dict: &mut Dictionary, key: &[u8], value: &Option<String>) {
        match value {
            Some(v) if v.is_empty() => { dict.remove(key); }
            Some(v) => { dict.set(key, Object::string_literal(v.as_str())); }
            None => {}
        }
    }

    apply_field(info, b"Title", &opts.title);
    apply_field(info, b"Author", &opts.author);
    apply_field(info, b"Subject", &opts.subject);
    apply_field(info, b"Keywords", &opts.keywords);
    apply_field(info, b"Creator", &opts.creator);

    Ok(doc)
}

pub async fn run(app: AppHandle, req: ProcessRequest) -> Result<PathBuf> {
    let op_id = req.operation_id.clone();

    let input = req.input_paths.into_iter().next()
        .ok_or_else(|| AppError::Validation("Edit Metadata requires one input file".into()))?;

    crate::pipeline::validate::validate_pdf(&input)?;

    let opts: MetadataOptions = serde_json::from_value(req.options.clone())
        .unwrap_or_default();

    emit_progress(&app, &op_id, 20, "Loading PDF…");
    let doc = Document::load(&input)
        .map_err(|e| AppError::Pdf(format!("Cannot open PDF: {e}")))?;

    emit_progress(&app, &op_id, 50, "Applying metadata…");
    let updated = apply_metadata(doc, &opts)?;

    let stem = input.file_stem()
        .and_then(|s| s.to_str())
        .unwrap_or("document");
    let out_stem = format!("{stem}_metadata");
    let stage = TempStage::new()?;
    let out_path = stage.output_path(&out_stem, "pdf");

    emit_progress(&app, &op_id, 80, "Saving PDF…");
    updated.save(&out_path)
        .map_err(|e| AppError::Pdf(format!("Failed to save PDF: {e}")))?;

    emit_complete(&app, &op_id, &out_path);
    Ok(out_path)
}
```

```rust
use std::path::{Path, PathBuf};
use lopdf::{Document, Object, Stream, Dictionary};
use tauri::AppHandle;

use crate::error::{AppError, Result};
use crate::pipeline::{
    temp::TempStage,
    progress::{emit_progress, emit_complete},
};
use crate::tools::ProcessRequest;

// (Option structs and defaults from above)

/// Embed a diagonal text watermark on every page of the PDF.
///
/// Strategy: build a PDF content stream that draws rotated grey text at the
/// page centre, then prepend it to each page's content array. This avoids the
/// need for a separate XObject form and is compatible with lopdf 0.31.
pub fn watermark_pdf(input: &Path, output: &Path, opts: &WatermarkOptions) -> Result<()> {
    if opts.text.is_empty() {
        return Err(AppError::Validation("Watermark text must not be empty".into()));
    }
    if !(0.0..=1.0).contains(&opts.opacity) {
        return Err(AppError::Validation(
            format!("Opacity must be between 0.0 and 1.0, got {}", opts.opacity)
        ));
    }

    // Parse hex colour → r, g, b (0.0–1.0)
    let color_hex = opts.color.trim_start_matches('#');
    let r = u8::from_str_radix(&color_hex.get(0..2).unwrap_or("88"), 16).unwrap_or(0x88) as f32 / 255.0;
    let g = u8::from_str_radix(&color_hex.get(2..4).unwrap_or("88"), 16).unwrap_or(0x88) as f32 / 255.0;
    let b = u8::from_str_radix(&color_hex.get(4..6).unwrap_or("88"), 16).unwrap_or(0x88) as f32 / 255.0;

    let mut doc = Document::load(input)
        .map_err(|e| AppError::Pdf(format!("Cannot open PDF: {e}")))?;

    let page_ids: Vec<_> = doc.get_pages().keys().copied().collect();

    for &page_num in &page_ids {
        let page_id = doc
            .get_pages()
            .get(&page_num)
            .copied()
            .ok_or_else(|| AppError::Pdf("Page not found".into()))?;

        // Get page dimensions from MediaBox (default to A4 if missing)
        let (width, height) = {
            let page_obj = doc.get_object(page_id)
                .map_err(|e| AppError::Pdf(e.to_string()))?
                .as_dict()
                .map_err(|e| AppError::Pdf(e.to_string()))?;
            let media = page_obj.get(b"MediaBox")
                .and_then(|o| o.as_array().ok())
                .filter(|a| a.len() == 4)
                .map(|a| {
                    let w = a[2].as_float().unwrap_or(612.0);
                    let h = a[3].as_float().unwrap_or(792.0);
                    (w, h)
                })
                .unwrap_or((612.0, 792.0));
            media
        };

        let cx = width / 2.0;
        let cy = height / 2.0;

        // Build a content stream: save state, set colour+opacity, rotate 45°, draw text
        let content = format!(
            "q\n\
             {r:.3} {g:.3} {b:.3} rg\n\
             /GS1 gs\n\
             BT\n\
             /F1 {fs:.1} Tf\n\
             1 0 0 1 {cx:.1} {cy:.1} Tm\n\
             0.7071 0.7071 -0.7071 0.7071 {cx:.1} {cy:.1} Tm\n\
             ({text}) Tj\n\
             ET\n\
             Q\n",
            r = r, g = g, b = b,
            fs = opts.font_size,
            cx = cx, cy = cy,
            text = opts.text.replace('(', r"\(").replace(')', r"\)"),
        );

        // Add ExtGState for opacity into page resources
        let gs_dict = {
            let mut d = Dictionary::new();
            d.set("Type", Object::Name(b"ExtGState".to_vec()));
            d.set("ca", Object::Real(opts.opacity));
            d.set("CA", Object::Real(opts.opacity));
            d
        };
        let gs_id = doc.add_object(Object::Dictionary(gs_dict));

        // Ensure Resources/ExtGState on page
        {
            let page_obj = doc.get_object_mut(page_id)
                .map_err(|e| AppError::Pdf(e.to_string()))?
                .as_dict_mut()
                .map_err(|e| AppError::Pdf(e.to_string()))?;
            let resources = page_obj
                .get_or_insert_with(b"Resources", || Object::Dictionary(Dictionary::new()))
                .as_dict_mut()
                .map_err(|e| AppError::Pdf(e.to_string()))?;
            let ext_gstate = resources
                .get_or_insert_with(b"ExtGState", || Object::Dictionary(Dictionary::new()))
                .as_dict_mut()
                .map_err(|e| AppError::Pdf(e.to_string()))?;
            ext_gstate.set("GS1", Object::Reference(gs_id));
        }

        // Prepend watermark stream to page Contents
        let stream = Stream::new(Dictionary::new(), content.into_bytes());
        let stream_id = doc.add_object(stream);

        let page_obj = doc.get_object_mut(page_id)
            .map_err(|e| AppError::Pdf(e.to_string()))?
            .as_dict_mut()

            }
            _ => {
                page_obj.set("Contents", Object::Array(vec![Object::Reference(stream_id)]));
            }
        }
    }
```

```rust
use lopdf::{Document, Object, Stream, Dictionary};
use tauri::AppHandle;

use crate::error::{AppError, Result};
use crate::pipeline::{
    temp::TempStage,
    progress::{emit_progress, emit_complete},
};
use crate::tools::ProcessRequest;

// (Option structs, Position enum, format_label from above)

pub fn add_page_numbers(input: &Path, output: &Path, opts: &PageNumberOptions) -> Result<()> {
    let mut doc = Document::load(input)
        .map_err(|e| AppError::Pdf(format!("Cannot open PDF: {e}")))?;

    let page_ids: Vec<_> = doc.get_pages().keys().copied().collect();
    let total = page_ids.len() as u32;

    for (idx, &page_num) in page_ids.iter().enumerate() {
        let n = opts.start + idx as u32;
        let label = format_label(&opts.format, n, total + opts.start - 1);

        let page_id = doc.get_pages()[&page_num];

        let (width, height) = {
            let page = doc.get_object(page_id).unwrap().as_dict().unwrap();
            page.get(b"MediaBox")
                .and_then(|o| o.as_array().ok())
                .filter(|a| a.len() == 4)
                .map(|a| (a[2].as_float().unwrap_or(612.0), a[3].as_float().unwrap_or(792.0)))
                .unwrap_or((612.0, 792.0))
        };

        let margin = 20.0_f32;
        let (x, y) = match opts.position {
            Position::BottomCenter => (width / 2.0 - 30.0, margin),
            Position::BottomRight  => (width - margin - 60.0, margin),
            Position::BottomLeft   => (margin, margin),
            Position::TopCenter    => (width / 2.0 - 30.0, height - margin - opts.font_size),
        };

        let content = format!(
            "q\n0 0 0 rg\nBT\n/F1 {fs:.1} Tf\n{x:.1} {y:.1} Td\n({label}) Tj\nET\nQ\n",
            fs = opts.font_size,
            x = x,
            y = y,
            label = label.replace('(', r"\(").replace(')', r"\)"),
        );

        let stream = Stream::new(Dictionary::new(), content.into_bytes());
        let stream_id = doc.add_object(stream);

        let page = doc.get_object_mut(page_id)
            .map_err(|e| AppError::Pdf(e.to_string()))?
            .as_dict_mut()
            .map_err(|e| AppError::Pdf(e.to_string()))?;

        match page.get(b"Contents").ok().map(|o| o.clone()) {
            Some(Object::Reference(id)) => {
                page.set("Contents", Object::Array(vec![Object::Reference(id), Object::Reference(stream_id)]));
            }
            Some(Object::Array(mut arr)) => {
                arr.push(Object::Reference(stream_id));
                page.set("Contents", Object::Array(arr));
            }
            _ => {
                page.set("Contents", Object::Array(vec![Object::Reference(stream_id)]));
            }
        }
    }

    doc.save(output)
        .map_err(|e| AppError::Pdf(format!("Failed to save PDF: {e}")))?;
    Ok(())
}
```

Apply black rectangle redactions over specified page regions. This is a **burn-in** redaction (not reversible): each region is drawn as a filled black box and the content stream is prepended so the box renders over the original content.

**Important caveat:** True forensic redaction requires removing the underlying content objects as well as drawing a black box. This v0.1 implementation provides visual redaction by drawing filled rectangles. For forensic-grade redaction, a future plan should use `qpdf` or `pdfcpu` to also strip the underlying content. Document this limitation clearly in code comments and in the UI.

```rust
use std::path::{Path, PathBuf};
use lopdf::{Document, Object, Stream, Dictionary};
use tauri::AppHandle;

use crate::error::{AppError, Result};
use crate::pipeline::{
    temp::TempStage,
    progress::{emit_progress, emit_complete},
};
use crate::tools::ProcessRequest;

// (RedactRegion and RedactOptions from above)

/// NOTE: This is visual redaction only. Content objects beneath the black
/// rectangle are NOT removed from the PDF byte stream. This means the
/// redaction can potentially be reversed by a knowledgeable user using a
/// hex editor or pdf parser. For forensic-grade redaction, the underlying
/// content must also be stripped (future work: integrate qpdf --qdf mode).
pub fn redact_pdf(input: &Path, output: &Path, opts: &RedactOptions) -> Result<()> {
    if opts.regions.is_empty() {
        return Err(AppError::Validation("Redact requires at least one region".into()));
    }

    // Validate all regions before processing
    for (i, r) in opts.regions.iter().enumerate() {
        if r.width <= 0.0 || r.height <= 0.0 {
            return Err(AppError::Validation(
                format!("Region {} has zero or negative dimension (width={}, height={})", i + 1, r.width, r.height)
            ));
        }
    }

    let mut doc = Document::load(input)
        .map_err(|e| AppError::Pdf(format!("Cannot open PDF: {e}")))?;

    let page_map: std::collections::HashMap<u32, _> = doc.get_pages().clone();

    for region in &opts.regions {
        let page_id = match page_map.get(&region.page) {
            Some(&id) => id,
            None => continue, // silently skip pages that don't exist
        };

        // PDF coordinate: (x, y) is bottom-left of rectangle
        let content = format!(
            "q\n0 0 0 rg\n{x:.2} {y:.2} {w:.2} {h:.2} re\nf\nQ\n",
            x = region.x,
            y = region.y,
            w = region.width,
            h = region.height,
        );

        let stream = Stream::new(Dictionary::new(), content.into_bytes());
        let stream_id = doc.add_object(stream);

        let page = doc.get_object_mut(page_id)
            .map_err(|e| AppError::Pdf(e.to_string()))?
            .as_dict_mut()
            .map_err(|e| AppError::Pdf(e.to_string()))?;

        // Append redaction stream AFTER existing content so it renders on top
        match page.get(b"Contents").ok().map(|o| o.clone()) {
            Some(Object::Reference(id)) => {
                page.set("Contents", Object::Array(vec![Object::Reference(id), Object::Reference(stream_id)]));
            }
            Some(Object::Array(mut arr)) => {
                arr.push(Object::Reference(stream_id));
                page.set("Contents", Object::Array(arr));
            }
            _ => {
                page.set("Contents", Object::Array(vec![Object::Reference(stream_id)]));
            }
        }
    }

    doc.save(output)
        .map_err(|e| AppError::Pdf(format!("Failed to save redacted PDF: {e}")))?;
    Ok(())
}

pub async fn run(app: AppHandle, req: ProcessRequest) -> Result<PathBuf> {
    let op_id = req.operation_id.clone();

    let input = req.input_paths.into_iter().next()
        .ok_or_else(|| AppError::Validation("Redact requires one input file".into()))?;

    crate::pipeline::validate::validate_pdf(&input)?;

    let opts: RedactOptions = serde_json::from_value(req.options.clone())
        .map_err(|e| AppError::Validation(format!("Invalid redact options: {e}")))?;

    emit_progress(&app, &op_id, 20, "Loading PDF…");

    let stem = input.file_stem().and_then(|s| s.to_str()).unwrap_or("document");
    let out_stem = format!("{stem}_redacted");
    let stage = TempStage::new()?;
    let out_path = stage.output_path(&out_stem, "pdf");

    emit_progress(&app, &op_id, 60, "Applying redactions…");
    redact_pdf(&input, &out_path, &opts)?;

    emit_complete(&app, &op_id, &out_path);
    Ok(out_path)
}
```

```svelte
  import { open } from '@tauri-apps/plugin-dialog';
  import { invoke } from '@tauri-apps/api/core';
  import { nanoid } from 'nanoid';
  import type { ProcessRequest } from '$lib/types';

  interface Region {
    id: string;
    page: number;
    x: number;
    y: number;
    width: number;
    height: number;
  }

  let filePath = $state<string | null>(null);
  let regions = $state<Region[]>([]);
  let running = $state(false);
  let result = $state<string | null>(null);
  let error = $state<string | null>(null);

  async function pickFile() {
    const selected = await open({
      multiple: false,
      filters: [{ name: 'PDF', extensions: ['pdf'] }],
    });
    if (typeof selected === 'string') {
      filePath = selected;
      result = null;
      error = null;
    }
  }

  function addRegion() {
    // Immutable: create new array
    regions = [...regions, { id: nanoid(), page: 1, x: 100, y: 100, width: 200, height: 30 }];
  }

  function removeRegion(id: string) {
    regions = regions.filter(r => r.id !== id);
  }

  function updateRegion(id: string, field: keyof Omit<Region, 'id'>, value: number) {
    // Immutable: map to new objects
    regions = regions.map(r => r.id === id ? { ...r, [field]: value } : r);
  }

  const canRun = $derived(filePath !== null && regions.length > 0 && !running);

  async function run() {
    if (!filePath || regions.length === 0) return;
    running = true;
    result = null;
    error = null;
    try {
      const request: ProcessRequest = {
        operation_id: nanoid(),
        tool: 'redact',
        input_paths: [filePath],
        output_stem: 'redacted',
        options: {
          regions: regions.map(({ page, x, y, width, height }) => ({ page, x, y, width, height })),
        },
      };
      result = await invoke<string>('process_pdf', { request });
    } catch (e: unknown) {
      error = e instanceof Error ? e.message : String(e);
    } finally {
      running = false;
    }
  }
```
